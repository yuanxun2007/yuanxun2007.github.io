Objective-C中的布尔类型(BOOL)

基础学习

布尔类型用来声明存储真值和假值的变量。在Objective-C中，类型BOOL,具有YES和NO两个值。

1. 首先来看看定义，在objc.h(ios sdk 8.4及osx sdk 10.10)文件中，我们看到如下定义：

#define OBJC_BOOL_DEFINED

/// Type to represent a boolean value.
#if !defined(OBJC_HIDE_64) && TARGET_OS_IPHONE && __LP64__
typedef bool BOOL;
#else
typedef signed char BOOL; 
// BOOL is explicitly signed so @encode(BOOL) == "c" rather than "C" 
// even if -funsigned-char is used.
#endif

#if __has_feature(objc_bool)
#define YES __objc_yes
#define NO  __objc_no
#else
#define YES ((BOOL)1)
#define NO  ((BOOL)0)
#endif

从上面的文件中，可以梳理出2点：

  • BOOL的定义，在64位的ios和32位ios、mac os，定义是不一样的。在64位ios中，它就是一个bool类型;在32位ios、mac os中，则是一个8位带符号的字符类型。
  • BOOL的值也有2种情况，在64位ios，就是YES和NO, 在32位ios、mac os，1表示YES，其他表示NO

2. 再来看一下使用

从上面我们知道，BOOL只有在64位ios中才是真正的布尔类型值，在32位ios及mac os中，编译器将BOOL认做8位二进制数，YES和NO值只是一个约定。所以使用的时候就需要注意以下2点：

  • 在32位ios及mac os中，非零值未必是BOOL的YES，但是0 一定是NO。所以最好不要直接的跟YES进行比较判断真假，这么做的话，可能会得到意想不到的结果
  • 在32位ios及mac os中，不小心的将长于1字节整型值赋给了BOOL变量，就只有低位会起作用（例如8960，写成16进制，就是0x2300,赋给BOOL变量，BOOL值将会是0，就是NO值）。

我建了2个工程进行测试验证，执行下面的代码及结果如下：

#import <Foundation/Foundation.h>

BOOL different (int a, int b) {
    return a - b;
}

int main(int argc, const char * argv[]) {
    @autoreleasepool {
        // insert code here...
        //第一组，直接看函数返回的真假值
        if (different(11, 10)) {
            printf ("11 != 10\n");
        } else {
            printf ("11 == 10\n");
        }

        if (different(10, 11)) {
            printf ("10 != 11\n");
        } else {
            printf ("10 == 11\n");
        }

        if (different(10, 10)) {
            printf ("10 != 10\n");
        } else {
            printf ("10 == 10\n");
        }

        if (different(12, 10)) {
            printf ("12 != 10\n");
        } else {
            printf ("12 == 10\n");
        }

        if (different(512, 256)) {
            printf ("512 != 256\n");
        } else {
            printf ("512 == 256\n");
        }

        printf ("***************************************\n");

        //第二组，函数返回的真假值，直接与YES的比较
        if (different(11, 10) == YES) {
            printf ("11 != 10\n");
        } else {
            printf ("11 == 10\n");
        }

        if (different(10, 11) == YES) {
            printf ("10 != 11\n");
        } else {
            printf ("10 == 11\n");
        }

        if (different(10, 10) == YES) {
            printf ("10 != 10\n");
        } else {
            printf ("10 == 10\n");
        }

        if (different(12, 10) == YES) {
            printf ("12 != 10\n");
        } else {
            printf ("12 == 10\n");
        }

        if (different(512, 256) == YES) {
            printf ("512 != 256\n");
        } else {
            printf ("512 == 256\n");
        }

        printf ("***************************************\n");

        //第三组，把int值赋给布尔变量，看真假
        BOOL x = 8960;
        if (x) {
            printf ("x is true\n");
        } else {
            printf ("x is false\n");
        }

        BOOL y = 1;
        if (y) {
            printf ("y is true\n");
        } else {
            printf ("y is false\n");
        }

        BOOL z = 255;
        if (z) {
            printf ("z is true\n");
        } else {
            printf ("z is false\n");
        }

        printf ("***************************************\n");

        //第四组，把int值赋给布尔变量，与YES的比较
        BOOL a = 8960;
        if (a == YES) {
            printf ("a == YES\n");
        } else {
            printf ("a == NO\n");
        }

        BOOL b = 1;
        if (b == YES) {
            printf ("b == YES\n");
        } else {
            printf ("b == NO\n");
        }

        BOOL c = 255;
        if (c == YES) {
            printf ("c == YES\n");
        } else {
            printf ("c == NO\n");
        }
    }
    return 0;
}
